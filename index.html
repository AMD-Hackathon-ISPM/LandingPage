# Persiapan Awal: Download Hexy core stack
$ git clone https://github.com/AMD-Hackathon-ISPM/hexy
$ cd hexy

# METODE 1: Dengan Docker (Default)
$ docker compose up --build
terminal - local dev
# Persiapan Awal: Download Hexy core stack
$ git clone https://github.com/AMD-Hackathon-ISPM/hexy
$ cd hexy

# METODE 2: Tanpa Docker (Local Development)
# Terminal 1 (Backend):
$ cd backend
$ pip install -r requirements.txt
$ uvicorn app.main:app --reload
# serves on http://localhost:8000

# Terminal 2 (Frontend):
$ cd frontend
$ npm install
$ npm run dev
# serves on http://localhost:5173
